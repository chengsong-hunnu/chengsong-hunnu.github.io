http模块

这个模块就是用来创建一个web服务器

const http = require('http')
const server = http.createServer()

server.on('request',(req,res)=>{
    if (req.url == '/cs') {
        console.log('cs')
        res.end('蒋晟松') // express框架提供了 res.send('data') 这个api
    } else if (req.url == '/') {
        res.end('123')
    } else {
        res.end('error')
    }

    // 这里的req就是客户端请求对象，包含url，query这些的东西
    // 这里的res就是服务器对客户端的响应对象，
    // 设置响应头信息，用于解决响应时中文乱码的问题    res.setHeader('Content-Type','text/html;charset=utf-8')
    // 设置状态码    res.statusCode = 200
    // 发送数据 res.end()
})
// 启动服务，需要给一个端口
server.listen(8000, () => {
    console.log('服务启动')
})

res.end() 传递的数据只能是字符串，res.send() 可传递对象这些，express会自动将其转为json返回

中间件

express中，当一个请求到达的服务器之后，可以在给客户响应之前连续调用多个中间件，来对本次请求和返回响应数据进行处理。

分类

  • 内置中间件 无需npm安装 如 express.static()
  • 第三方中间件 通过npm安装 如 body-parser
  • 自定义中间件

也可分为：

  • 应用中间件
    • 全局使用 app.use() 所有路由都生效
    • 局部使用 app.请求方法(地址,[中间件…],回调函数) 当前路由生效
  • 路由级别中间件,只是绑定在router上
    • router.use(中间件)
    • router.请求方法(地址,[中间件…],回调函数)

内置中间件

express.static() 静态资源管理

express.json 接收json格式提交的数据

app.use(express.json())

其在接收完数据之后，会将数据的对象形式挂载到请求对象的body属性上

express.urlencoded 处理post表单提交的数据

app.use(express.urlencoded({ extended: false }))

其在接收完数据之后，也会将数据的对象形式挂载到请求对象的body属性上

上面两个需要的express版本需要>=4.16.0 所以会有一些兼容性的问题

上面两个不会出现覆盖的问题，因为传递的数据只会是一种格式

  • extended: false：表示使用系统模块querystring来处理，会剔除传递过来的方法和对象，也是官方推荐的
  • extended: true：表示使用第三方模块qs来处理

第三方中间件

以body-parser为例

使用前需要安装

npm install -S body-parser

然后在要使用的文件里面引入，并使用app.use()调用，然后就行了

var bodyParser = require('body-parser')
app.use(bodyParser.urlencoded({ extended: true }))

自定义中间件

自定义中间件，其本质就是定义一个处理请求的函数，只是此函数中除了有request和response参数外还必须包含一个next参数，此参数作用让中间件能够让流程向下执行下去直到匹配到的路由中发送响应给客户端。也可以通过给request对象添加属性来进行中间件数据的向下传递

function myfn(req,res,next
    ……
    next()    // 中间件最后一句必须执行此语句，进行放行
}

// 使用
app.use(myfn)

在整个请求链路中，所有中间件与最终路由共用一份req和res

注意上面的next() 有可能与异步代码发生冲突，所以注意位置

异常中间件

作用:专门用来捕获整个项目发生的异常错误，从而防止项目异常崩溃的问题产生(友好显示异常)。

格式:错误级别中间件的函数参数中，必须有四个形参，分别是(err,req,res , next)

app.get('/err', (req, res) => {
    throw new Error('发生错误')
})

app.use((err, req, res, next) => {
    console.log('err:', err)
    res.send('error:' + err.message) // error:发生错误
})

404中间件

如果所有的路由都匹配不上，即转到这个页面

app.use((req, res, next) => {
    res.status(404).send('404')
})

上面两个中间件的app.use() 都需要放在代码最下面，不然会使代码产生错误

session

因为cookie是存放在客户端里的，所以会有被修改的风险，所以需要存放在服务端的session，然后在cookie里面存一个session的id

使用 cookie-session 这个中间件

var session = require('cookie-session')

app.use(session({
    name:'sessionId',    // 这就是存在cookie里的cookie名
    secret:'cscscscsca',    // 这的secret是对sessionid进行加密，防止伪造
    maxAge:20*60*1000,
    rolling:true    // 让时间滚动刷新
}))

app.get('/',(req,res)=>{
    if (req.session['count']) {
        req.session['count'] += 1
        // 这里就获取到session值
        res.send({ cookie: 'cookie页面', count: req.session['count'] })
    } else {
        res.send('no cookie of count')
        // 这里就创建一个session值
        req.session['count'] = 1
    }
})

连接数据库

数据库的类型这里就先记录一个mysql吧

需要安装一个mysql的中间件，然后进行连接，进行crud

//连接mysql数据库的文件
const mysql = require("mysql");

module.exports = {
    //数据库配置
    config: {
        host: "localhost", // 数据库的域名和端口
        port: "3306",
        user: "root",
        password: "xxxxx",
        database: "exapp"    // 数据库名
    },
    // 连接数据库，使用mysql的连接池
    // 连接池的对象
    sqlConnect: function (sql, sqlArr, callBack) {
        let pool = mysql.createPool(this.config);
        // 每一次连接查询完后都需要关闭连接
        pool.getConnection((err, conn) => {
            console.log("数据库连接");
            if (err) {
                console.log("连接失败");
                return;
            }
            // 事件驱动回调,执行数据库操作
            conn.query(sql, sqlArr, callBack);
            // 释放连接
            conn.release();
        });
    },
    // 操纵user信息的promise回调
    asyncSqlconnect: function (asyncSql, sqlArr) {
        return new Promise((resolve, reject) => {
            let pool = mysql.createPool(this.config);
            pool.getConnection((err, conn) => {
                console.log("数据库连接", asyncSql);
                if (err) {
                    reject(err);
                } else {
                    conn.query(asyncSql, sqlArr, (err, data) => {
                        if (err) {
                            reject(err);
                        } else {
                            // 将异步操作的结果作为参数传递出去
                            resolve(data);
                        }
                    });
                    // 释放连接
                    conn.release();
                }
            });
        }).catch((err) => {
            console.log("catch", err);
        });
    }
};

// upload.js
const express = require('express');
const router = express.Router();

let upload = multer({
    // dest: 'uploads/' 最简单的方式，只指定一下存储的位置
    // 这里的storage是控制文件存储在哪里，这里是存在磁盘上，还有存在内存里的，但不使用
    storage: multer.diskStorage({
        destination: function (req, file, cb) {
            // 这里是地址
            cb(null, './uploads/');
        },
        // 用于确定存储在上面目录里的文件的文件名
        filename: function (req, file, cb) {
            // 这里对图片名加上时间戳

        }
    })
});

接口设计

前后端分离式开发需要进行数据交互，传输的数据被偷窥、被抓包、被伪造时有发生，需要设计一套比较安全的API接口方案

并不是所有的接口都需要考虑安全的，有些接口是公开的，任何人只要知道地址都可以调用，对于一些项目中需要用户登录才能访问的接口才需要考虑安全问题。

一般解决的方案有以下几类:

  • token令牌认证[jwt json web token]
  • AK (app key)&SK (secret key)【用户名&密码】
  • 时间戳超时验证+签名算法字符串
  • URL签名(算法，非对称算法)·数据脱敏(防范数据库数据泄露)
  • HTTPS
    • 数字证书(防运营商)
  • iP黑/白名单(服务器层面的限制，apache、nginx)
  • oAuth2.0

以jwt为例

jwt

使用JWT进行鉴权的工作流程如下(重点):

  • 用户使用用户名密码来请求服务器
  • 服务器进行验证用户的信息(查数据库)
  • 服务器通过验证发送给用户一个token(令牌)
  • 客户端存储token (Vuex+localStorage)，并在每次请求时附送上这个token值
  • 服务端验证token值，并返回数据

jwt由 头部、载荷、签名三部分构成

在nodejs里面使用jwt的话，需要安装一个中间件

npm install jsonwebtoken -S
const jwt = require('jsonwebtoken');
let secret = 'token'
const Token = {
  encrypt:function(data,time){ //data需加密数据，time到期时间
    // secret加密密钥，expiresIn到期时间，issuer发行人
    return jwt.sign(data, secret, {expiresIn:time,issuer:'cs'})
  },
  decrypt:function(token){
    try {
      let data = jwt.verify(token, secret);
      return {
        token:true,
        id:data.id
      };
    } catch (e) {
      return {
        token:false,
        data:e
      }
    }
  }
}
module.exports = Token;
const Token = require('../utils/token')

//code……

    //有效token

    //无效token
}

// upload.js
const path = require('path')
const fs = require('fs')

const uploadimg = (ctx) => {
    console.log(JSON.stringify(ctx.request, null, ' '))
    let remotefilePath = null
    if (ctx.request.files['file']) {
        // 创建可读流，同时在ctx.request.files里面选出属性名为file的文件的path创建流
        const reader = fs.createReadStream(ctx.request.files['file']['path'])
        // 创建一个文件保存在哪里的路径
        let time = new Date().getTime()
        let filePath = `${path.resolve(__dirname, '../upload')}/${time}${ctx.request.files['file']['name']}`
        // 这就是返回给前端进行图片访问的url，需要配合前面设置的静态资源目录路径
        remotefilePath = `http://127.0.0.1/upload/${time}${ctx.request.files['file']['name']}`
        // 创建可写流，即保存文件
        const upStream = fs.createWriteStream(filePath)
        // 可读流通过管道写入可写流
        reader.pipe(upStream)
    }
    return remotefilePath
}

cookie

ctx.cookies.get(name, [options])

获得 cookie 中名为 name 的值，options 为可选参数

ctx.cookies.set(name, value, [options])
ctx.cookies.set('cookie', 'setCookie', {maxAge:10*60*1000,})

设置 cookie 中名为 name 的值，options 为可选参数：

  • maxAge 一个数字，表示 Date.now()到期的毫秒数
  • signed 是否要做签名
  • expires cookie有效期
  • pathcookie 的路径，默认为 /' ，就是可访问这个cookie的路径
  • domain cookie 的域，默认当前域下面所有的页面都能访问
  • secure false 表示 cookie 通过 HTTP 协议发送，true 表示 cookie 通过 HTTPS 发送。
  • httpOnly true 表示 cookie 只能通过 HTTP 协议发送
  • overwrite 一个布尔值，表示是否覆盖以前设置的同名的Cookie（默认为false）。 如果为true，在设置此cookie时，将在同一请求中使用相同名称（不管路径或域）设置的所有Cookie将从Set-Cookie头部中过滤掉。

注意：Koa 使用了 Express 的 cookies 模块，options 参数只是简单地直接进行传递。

而且在 cookie 里面有一个bug或特性，不能直接设置中文，会出现错误。

只能将中文转为base64的格式进行存储，在使用的时候再转回来

ctx.cookies.set('cs', new Buffer('晟松').toString('base64'))
ctx.body = new Buffer(ctx.cookies.get('cs'), 'base64').toString()

但好像这里面的Buffer已经被弃用了，所以能不用就不用
